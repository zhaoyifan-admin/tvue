代码示例

<!-- 基础用法 -->
<tvue-ant-cascader
  v-model="value"
  :lazy="true"
  :lazy-load="lazyLoadMethod"
  :dic="[]"  <!-- 重要：设置为空数组触发懒加载 -->
  placeholder="请选择"
/>

<!-- 多选模式 -->
<tvue-ant-cascader
  v-model="value"
  :lazy="true"
  :lazy-load="lazyLoadMethod"
  :dic="[]"
  :multiple="true"
/>

<!-- 可搜索模式 -->
<tvue-ant-cascader
  v-model="value"
  :lazy="true"
  :lazy-load="lazyLoadMethod"
  :dic="[]"
  :filterable="true"
/>

<!-- lazyLoad 方法签名 -->
lazyLoadMethod(node, resolve) {
  // node.level - 当前层级（从0开始）
  // node.data - 当前节点数据
  // resolve(list) - 回调函数，传入子节点数组
  
  if (node.level === 0) {
    // 加载第一层数据
    resolve([{
      value: '1',
      label: '选项1',
      leaf: false  // false表示还有子节点
    }]);
  }
}
